click button to show name
        

            const nameContainer = document.querySelector(".name");
            const showNameBtn = document.querySelector("button");

            let isNameShowing = false;

            showNameBtn.addEventListener("click", () => {
              isNameShowing = !isNameShowing;
              showNameBtn.textContent = `${
                isNameShowing 
                  ? "Hide Name" 
                  : "Show Name"
              }`;
              nameContainer.textContent = `${
                isNameShowing 
                  ? "Ebenezer Ogungbe" 
                  : "click button to show name"
              }`;
